### 6. 添加点击事件（第962行之后，graphChart.setOption(option); 之后）

在 `graphChart.setOption(option);` 之后添加：

```typescript
// 点击节点事件 - 定位到对应代码
graphChart.on('click', function(params: any) {
  if (params.dataType === 'node') {
    const userUuid = params.data.id;
    const userName = params.data.name;
    scrollToMemberCard(userUuid, userName);
  }
});
```

### 7. 添加滚动函数（在 initGraphChart 函数之后）

添加以下函数：

```typescript
// 滚动到指定成员卡片
const scrollToMemberCard = (userUuid: string, userName: string) => {
  const cardId = `member-card-${userUuid}`;
  const cardElement = document.getElementById(cardId);
  
  if (cardElement) {
    // 添加高亮效果
    cardElement.style.transition = 'all 0.3s ease';
    cardElement.style.boxShadow = '0 0 20px rgba(64, 158, 255, 0.8)';
    cardElement.style.transform = 'scale(1.02)';
    
    // 滚动到该卡片
    cardElement.scrollIntoView({
      behavior: 'smooth',
      block: 'center',
    });
    
    // 3秒后移除高亮效果
    setTimeout(() => {
      cardElement.style.boxShadow = '';
      cardElement.style.transform = '';
    }, 3000);
    
    // 显示提示
    ElNotification({
      title: '已定位',
      message: `已定位到用户 ${userName || userUuid} 的代码`,
      type: 'success',
      duration: 2000,
    });
  }
};
```
